15// Global semaphore to limit concurrent file operations
16static HASH_SEMAPHORE: std::sync::OnceLock<Arc<Semaphore>> = std::sync::OnceLock::new();
17
18fn get_hash_semaphore(max_concurrent: usize) -> &'static Arc<Semaphore> {
19    HASH_SEMAPHORE.get_or_init(|| Arc::new(Semaphore::new(max_concurrent)))
20}

85/// Calculate all configured hashes for a file
86pub async fn calculate_all_hashes_with_config(path: &Path, config: &HashConfig) -> Result<Hashes> {
87    let path = path.to_path_buf();
88    let semaphore = get_hash_semaphore(config.max_concurrent_operations);
89
90    // Acquire permit to limit concurrent operations
91    let _permit = semaphore
92        .acquire()
93        .await
94        .map_err(|_| HashError::SemaphoreError)?;
95
96    let mut tasks = Vec::new();
97    let buffer_size = config.buffer_size;
98
99    if config.algorithms.md5 {
100        let path_clone = path.clone();
101        tasks.push(task::spawn_blocking(move || {
102            calculate_md5_sync(&path_clone, buffer_size).map(Some)
103        }));
104    } else {
105        tasks.push(task::spawn_blocking(|| Ok(None)));
106    }
107
108    if config.algorithms.sha256 {
109        let path_clone = path.clone();
110        tasks.push(task::spawn_blocking(move || {
111            calculate_sha256(&path_clone, buffer_size).map(Some)
112        }));
113    } else {
114        tasks.push(task::spawn_blocking(|| Ok(None)));
115    }
116
117    if config.algorithms.sha512 {
118        let path_clone = path.clone();
119        tasks.push(task::spawn_blocking(move || {
120            calculate_sha512(&path_clone, buffer_size).map(Some)
121        }));
122    } else {
123        tasks.push(task::spawn_blocking(|| Ok(None)));
124    }
125
126    if config.algorithms.blake3 {
127        let path_clone = path.clone();
128        tasks.push(task::spawn_blocking(move || {
129            calculate_blake3(&path_clone, buffer_size).map(Some)
130        }));
131    } else {
132        tasks.push(task::spawn_blocking(|| Ok(None)));
133    }
134
135    let results: Vec<_> = futures::future::try_join_all(tasks).await?;
136
137    Ok(Hashes {
138        md5: results[0].as_ref().ok().and_then(|v| v.clone()),
139        sha256: results[1].as_ref().ok().and_then(|v| v.clone()),
140        sha512: results[2].as_ref().ok().and_then(|v| v.clone()),
141        blake3: results[3].as_ref().ok().and_then(|v| v.clone()),
142    })
143}

145/// Calculate all hashes with default configuration
146pub async fn calculate_all_hashes(path: &Path) -> Result<Hashes> {
147    let config = HashConfig::default();
148    calculate_all_hashes_with_config(path, &config).await
149}

151/// Calculate MD5 hash asynchronously
152pub async fn calculate_md5(path: &Path) -> Result<String> {
153    let path = path.to_path_buf();
154    let semaphore = get_hash_semaphore(10);
155    let _permit = semaphore
156        .acquire()
157        .await
158        .map_err(|_| HashError::SemaphoreError)?;
159
160    task::spawn_blocking(move || calculate_md5_sync(&path, 8192)).await?
161}

163fn calculate_md5_sync(path: &Path, buffer_size: usize) -> Result<String> {
164    let file = File::open(path)?;
165    let mut reader = BufReader::new(file);
166    let mut hasher = Md5::new();
167    let mut buffer = vec![0; buffer_size];
168
169    loop {
170        let count = reader.read(&mut buffer)?;
171        if count == 0 {
172            break;
173        }
174        hasher.update(&buffer[..count]);
175    }
176
177    Ok(format!("{:x}", hasher.finalize()))
178}
